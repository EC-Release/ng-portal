type Agent struct {
    Id            string
    Conn          *WSocket       `json:"-"`
    Options       *config.Config `json:"-"`
    SessionIdChan chan string    `json:"-"`
    SessionId     string         `json:",omitempty"`
    BindId        string         `json:",omitempty"`
    TimeCreated   time.Time
    UsageRate     int
    LastPong      time.Time `json:",omitempty"`
}

type FuseAgent ¶ 

type FuseAgent struct {
    //no need to ensure thread-safe. See https://golang.org/pkg/net/#Conn
    //sync.Mutex
    Id        string
    SessionId string         `json:",omitempty"`
    BindId    string         `json:",omitempty"`
    Options   *config.Config `json:"-"`
    //Conn          *net.TCPConn   `json:"-"`
    Conn        util.StreamIntr `json:"-"`
    TimeCreated time.Time
}
